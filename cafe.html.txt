    <script>
        let total = 0;

        function updateQty(id, change, price) {
            const qtyElement = document.getElementById(id);
            let currentQty = parseInt(qtyElement.innerText);
            let newQty = currentQty + change;
            if (newQty < 0) return;

            qtyElement.innerText = newQty;
            total += (change * price);
            document.getElementById('total-price').innerText = "₹" + total;
        }

        function placeOrder() {
            const name = document.getElementById('custName').value.trim();
            const mobile = document.getElementById('custMobile').value.trim();
            const address = document.getElementById('custAddress').value.trim();

            if (total === 0) {
                alert("Please add items to your cart first!");
                return;
            }
            if (!name || !mobile || !address) {
                alert("Please fill in your Name, Mobile, and Address!");
                return;
            }

            // Compile the item list
            let itemSummary = "";
            const menuItems = [
                { id: 'qty1', name: 'Bikaneri Samosa' },
                { id: 'qty2', name: 'Moongdal Kachori' },
                { id: 'qty3', name: 'Veg Momos' }
            ];

            menuItems.forEach(item => {
                const qty = document.getElementById(item.id).innerText;
                if (parseInt(qty) > 0) {
                    itemSummary += `• ${item.name} x ${qty}\n`;
                }
            });

            // Format WhatsApp Message
            const msg = `*NEW ORDER - CAFE BAITHAK*%0A` +
                        `--------------------------%0A` +
                        `*Customer:* ${name}%0A` +
                        `*Mobile:* ${mobile}%0A` +
                        `*Address:* ${address}%0A` +
                        `--------------------------%0A` +
                        `*Items:*%0A${encodeURIComponent(itemSummary)}` +
                        `--------------------------%0A` +
                        `*TOTAL BILL: ₹${total}*%0A` +
                        `--------------------------`;

            const whatsappNumber = "919538003340";
            window.open(`https://wa.me/${whatsappNumber}?text=${msg}`, '_blank');
        }
    </script>
